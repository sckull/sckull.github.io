function save_profile($player, $args) {
	global $pdo;
  	$params = ["player"=>$player];
	$setStr = "";
  	foreach ($args as $key => $value) {

	}

  	$stmt = $pdo->prepare("UPDATE players SET $setStr WHERE username = :player");

}
